Crea la Base de Datos y Tabla: Ejecuta las siguientes consultas en MySQL Workbench, phpMyAdmin, o desde la terminal:

CREATE DATABASE luchacriolla;

USE luchacriolla;

CREATE TABLE cartas (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(50) NOT NULL,
    nivel_poder INT NOT NULL
);

CREATE TABLE cartas_auditoria (
    id INT AUTO_INCREMENT PRIMARY KEY,
    carta_id INT NOT NULL,
    nombre VARCHAR(50),
    nivel_poder INT,
    fecha_registro DATETIME DEFAULT CURRENT_TIMESTAMP
);

DELIMITER $$

CREATE TRIGGER after_insert_cartas
AFTER INSERT ON cartas
FOR EACH ROW
BEGIN
    INSERT INTO cartas_auditoria (carta_id, nombre, nivel_poder)
    VALUES (NEW.id, NEW.nombre, NEW.nivel_poder);
END $$

DELIMITER ;

DELIMITER $$

CREATE PROCEDURE obtener_cartas_poder_alto(IN min_poder INT)
BEGIN
    SELECT * 
    FROM cartas
    WHERE nivel_poder > min_poder;
END $$

DELIMITER ;
